    1146 +    private func sortedRowsForGroup(_ rows: [TaskBoardRow], groupName: String) -> [TaskBoardRow] {

    1164 +    private func applyCustomOrder(_ rows: [TaskBoardRow], orderedIDs: [String]) -> [TaskBoardRow] {
    1165 +        let byID = Dictionary(uniqueKeysWithValues: rows.map { ($0.session.id, $0) })

    1168 +        for id in orderedIDs {
    1169 +            guard let row = byID[id] else { continue }
    1170 +            result.append(row)
    1171 +            used.insert(id)
    1172 +        }
    1173 +        for row in rows where !used.contains(row.session.id) {
    1174 +            result.append(row)
    1175 +        }
    1176 +        return result

    1319 +    private func handleTaskDrop(
    1320 +        destinationSessionID: String,
    1321 +        destinationGroupName: String,
    1322 +        destinationBucket: TaskSortBucket,
    1323 +        destinationIsPinnedGroup: Bool
    1324 +    ) -> Bool {
    1325 +        defer {
    1326 +            taskPanelDraggingSessionID = nil
    1327 +            taskPanelDraggingGroupName = nil
    1328 +            taskPanelDraggingBucket = nil
    1329 +        }
    1330 +        guard let draggingID = taskPanelDraggingSessionID,
    1331 +              let draggingGroup = taskPanelDraggingGroupName,
    1332 +              let draggingBucket = taskPanelDraggingBucket,
    1333 +              draggingID != destinationSessionID else {
    1334 +            return false
    1335 +        }
    1336 +
    1337 +        if destinationIsPinnedGroup {
    1338 +            guard taskPanelPinnedSessionIDs.contains(draggingID),
    1339 +                  taskPanelPinnedSessionIDs.contains(destinationSessionID),
    1340 +                  draggingGroup == destinationGroupName else {
    1341 +                return false
    1342 +            }
    1343 +            var ids = taskPanelPinnedOrder.filter { taskPanelPinnedSessionIDs.contains($0) }
    1344 +            guard let from = ids.firstIndex(of: draggingID),
    1345 +                  let to = ids.firstIndex(of: destinationSessionID) else {
    1346 +                return false
    1347 +            }
    1348 +            moveElement(in: &ids, from: from, to: to)
    1349 +            taskPanelPinnedOrder = ids
    1350 +            persistTaskPanelSortState()
    1351 +            return true
    1352 +        }
    1353 +
    1354 +        guard draggingGroup == destinationGroupName,
    1355 +              draggingBucket == destinationBucket,
    1356 +              destinationBucket != .pinned else {
    1357 +            return false
    1358 +        }
    1359 +
    1360 +        let key = taskGroupBucketStorageKey(groupName: destinationGroupName, bucket: destinationBucket)
    1361 +        var ids = taskPanelOrderByGroupBucket[key] ?? []
    1362 +        if !ids.contains(draggingID) {
    1363 +            ids.append(draggingID)
    1364 +        }
    1365 +        if !ids.contains(destinationSessionID) {
    1366 +            ids.append(destinationSessionID)
    1367 +        }
    1368 +        guard let from = ids.firstIndex(of: draggingID),
    1369 +              let to = ids.firstIndex(of: destinationSessionID) else {
    1370 +            return false
    1371 +        }
    1372 +        moveElement(in: &ids, from: from, to: to)
    1373 +        taskPanelOrderByGroupBucket[key] = ids
    1374 +        persistTaskPanelSortState()
    1375 +        return true
    1376 +    }

    1378 +    private func moveElement(in ids: inout [String], from: Int, to: Int) {
    1379 +        guard ids.indices.contains(from), ids.indices.contains(to), from != to else { return }
    1380 +        let item = ids.remove(at: from)
    1381 +        ids.insert(item, at: to)
